```
preprocess(){
t2t-trainer \
  --generate_data \
  --data_dir="$data" \
  --problems=summarize_cnn_dailymail32k \
  --model=transformer \
  --hparams_set=transformer_base_single_gpu \
  --output_dir="$root"
  }

train(){
t2t-trainer \
    --data_dir="$data" \
    --problems=summarize_cnn_dailymail32k \
    --model=transformer \
    --hparams_set=transformer_base_single_gpu \
    --output_dir="$root"
}

translate(){
t2t-decoder \
  --data_dir="$data" \
  --problems=summarize_cnn_dailymail32k \
  --model=transformer \
  --hparams_set=transformer_base_single_gpu \
  --output_dir="$root" \
  --decode_from_file "cnndm.test.src.50.txt" \
  --decode_to_file "pred.txt" \
  --decode_beam_size 10 \
  --batch_size 16
}
```
